47@contracts.check_contracts 

48class FourInARow: 

49 """ 

50 A class representing a game of Four-in-a-Row. 

51 

52 Attributes: 

53 - n: Size of the square grid (n x n) 

54 - p1_human: True if Player 1 is a human player 

55 - p2_human: True if Player 2 is a human player 

56 - result: outcome of a game 

57 - board: the current state of the board 

58 - p1_to_play: True if it is Player 1's turn to play 

59 

60 Representation Invariants: 

61 - self.result in (WIN, DRAW, LOSS, IN_PROGRESS) 

62 """

64 n: int 

65 p1_human: bool 

66 p2_human: bool 

67 result: int

71 def __init__(self, n: int, p1_human: bool, p2_human: bool) -> None: 

72 """ 

73 Initialize this game of four-in-a-row to be played on an n-by-n grid. 

74 

75 The game begins with a result of IN_PROGRESS and p1 to play. 

76 

77 <p1_human> and <p2_human> indicate whether each of p1 and p2 are human 

78 or computer players. 

79 

80 Preconditions: 

81 - n > 3 

82 """ 

83 self.n = n 

84 self.p1_human = p1_human 

85 self.p2_human = p2_human 

86 self.result = IN_PROGRESS 

87 self.board = Grid(n) 

88 self.p1_to_play = True 

89 

90 def play(self) -> None: 

91 """ 

92 Play this game and update the result attribute to indicate the outcome. 

93 

94 Preconditions: 

95 - self.result == IN_PROGRESS 

96 

97 Since this function has a lot of text output, we use a feature 

98 of doctest which allows us to conveniently ignore part of the output. 

99 The doctest:+ELLIPSIS comment tells doctest to ignore the ... and 

100 any output after it when checking for matching output. 

101 

102 >>> my_input = StringIO(GAME_SCRIPT_X_WINS) 

103 >>> sys.stdin = my_input 

104 >>> g = load_game() # doctest:+ELLIPSIS 

105 Enter game... 

106 >>> g.play() # doctest:+ELLIPSIS 

107 Initial Board... 

108 >>> g.result == WIN 

109 True 

110 """ 

111 i = 1 

112 print('Initial Board') 

113 print(self.board) 

114 last_player = self.p1_to_play 

115 while not self._take_turn(): 

116 print(f"Turn {i}") 

117 print(self.board) 

118 i += 1 

119 last_player = self.p1_to_play 

120 

121 if self.p1_to_play != last_player: # board became full! 

122 self.result = DRAW 

123 elif self.p1_to_play: # first player won! 

124 self.result = WIN 

125 else: # first player lost! 

126 self.result = LOSS 

127 print(f"Turn {i}") 

128 print(self.board) 

129 

130 def _take_turn(self) -> bool: 

131 """ 

132 Return True if this play has caused the game to end or False otherwise. 

133 

134 This could happen either by the board becoming full or the current 

135 player winning the game. 

136 

137 If the game has not ended, then whose turn it is will be updated. 

138 

139 Preconditions: 

140 - self.result == IN_PROGRESS 

141 

142 Note: # doctest:+ELLIPSIS is a feature of doctest used below, which 

143 tells doctest to treat ... as a wildcard match. For example, 

144 below where it says "X computer..." this means that the doctest 

145 will pass if the result of g._take_turn() prints "X computer" 

146 followed by anything. Since both players are computers, we 

147 can't know what the exact output will be, so using a wildcard 

148 match is convenient here. 

149 

150 >>> g = FourInARow(4, False, False) # a game between computer players 

151 >>> g._take_turn() # doctest:+ELLIPSIS 

152 X computer... 

153 >>> g.p1_to_play 

154 False 

155 >>> g._take_turn() # doctest:+ELLIPSIS 

156 O computer... 

157 >>> g.p1_to_play 

158 True 

159 """ 

160 # determine whose turn it is 

161 if self.p1_to_play: 

162 symbol = 'X' 

163 is_human = self.p1_human 

164 else: 

165 symbol = 'O' 

166 is_human = self.p2_human 

167 

168 # get the coordinate that the player chooses to play their piece at 

169 column_move = None 

170 row_move = None 

171 if is_human: # get input from the user 

172 while row_move is None: 

173 column_move = int(input(f"{symbol} to move: ")) 

174 print(column_move) 

175 row_move = self.board.drop(column_move, symbol) 

176 else: 

177 # random computer move 

178 print(f"{symbol} computer moving...", end="") 

179 column_moves = list(range(self.board.n)) 

180 shuffle(column_moves) 

181 while row_move is None: 

182 column_move = column_moves.pop() 

183 row_move = self.board.drop(column_move, symbol) 

184 print(column_move) 

185 # given where the move was made, check if this player won 

186 if self.board.has_fiar((row_move, column_move)): 

187 return True 

188 

189 self.p1_to_play = not self.p1_to_play # update whose turn it is 

190 # return whether the grid is full 

191 return self.board.is_full()

197def load_game() -> FourInARow: # pragma: no cover 

198 """ 

199 Return a four-in-a-row game. 

200 

201 Input from the user is read from standard input which specifies: 

202 n is_p1_human? is_p2_human? 

203 

204 >>> my_input = StringIO('4 True False\\n') 

205 >>> sys.stdin = my_input 

206 >>> g = load_game() # note: this will print the prompt for input, as below. 

207 Enter game parameters (<n> <is_p1_human?> <is_p2_human?>): 

208 >>> g.board.n 

209 4 

210 >>> g.p1_human 

211 True 

212 >>> g.p2_human 

213 False 

214 >>> g.p1_to_play 

215 True 

216 >>> g.result == IN_PROGRESS 

217 True 

218 """ 

219 input_string = input( 

220 "Enter game parameters (<n> <is_p1_human?> " "<is_p2_human?>):\n" 

221 ) 

222 # pull out the three pieces of information 

223 n, p1_human, p2_human = input_string.split() 

224 # return the actual game based on this information 

225 return FourInARow(int(n), p1_human == 'True', p2_human == 'True')

228def play_game(script: str | None = None) -> FourInARow: # pragma: no cover 

229 """ 

230 Play a game of four-in-a-row, returning the FourInARow object created. 

231 

232 If <script> is provided, it is used to read input from. 

233 Otherwise, the user will be prompted for the moves of the first player in 

234 a game against a computer on a 6-by-6 board. 

235 

236 See test_a0.py for a sample test. Since this function has a lot of text 

237 output, no doctest examples are provided. 

238 """ 

239 if script is not None: 

240 sys.stdin = StringIO(script) 

241 fiar = load_game() 

242 else: 

243 fiar = FourInARow(6, True, False) # play human versus computer 

244 

245 fiar.play() 

246 

247 if fiar.result == DRAW: 

248 print("Game ended in a draw!") 

249 elif fiar.result == WIN: 

250 print("Game won by X player!") 

251 else: # fiar.result == LOSS 

252 print("Game won by O player!") 

253 

254 return fiar
